      operation. Then [P] satisfies [ClosedUnderOp Œ±] iff
      for [x1 : A s1], [x2 : A s2], ..., [xn : A sn],

<<
      P s1 x1 ‚àß P s2 x2 ‚àß ... ‚àß P sn xn

    match w with
    | [:s:] => P s

                    ‚àÄ (x : A s), P s x ‚Üí ClosedUnderOp (Œ± x)
    end.

(** [P : ‚àÄ s, A s ‚Üí Type] is a subalgebra predicate if it is closed
    under operations [IsClosedUnderOps A P] and [P s x] is an h-prop. *)

Section subalgebra.
  Context
    {œÉ : Signature} (A : Algebra œÉ)
    (P : ‚àÄ s, A s ‚Üí Type) `{!IsSubalgebraPredicate A P}.

(** The subalgebra carriers is the family of subtypes defined by [P]. *)

  Definition carriers_subalgebra : Carriers œÉ
    := Œª (s : Sort œÉ), {x | P s x}.

(** Let [Œ± : A s1 ‚Üí ... ‚Üí A sn ‚Üí A t] be an operation and let [C :
    ClosedUnderOp A P Œ±]. The corresponding subalgebra operation
    [op_subalgebra Œ± C : (A&P) s1 ‚Üí ... ‚Üí (A&P) sn ‚Üí (A&P) t] is
    given by

<<
      op_subalgebra Œ± C (x1; p1) ... (xn; pn) =
        (Œ± x1 ... xn; C x1 p1 x2 p2 ... xn pn).
>>
*)

  Fixpoint op_subalgebra {w : SymbolType œÉ}
    : ‚àÄ (Œ± : Operation A w),
      ClosedUnderOp A P Œ± ‚Üí Operation carriers_subalgebra w
    := match w with
       | [:t:] => Œª Œ± c, (Œ±; c)
       | s ::: w' => Œª Œ± c x, op_subalgebra (Œ± x.1) (c x.1 x.2)
       end.

(** The subalgebra operations [ops_subalgebra] are defined in terms of
    [op_subalgebra]. *)

  Definition ops_subalgebra (u : Symbol œÉ)
    : Operation carriers_subalgebra (œÉ u)
    := op_subalgebra u.#A (is_closed_under_ops_subalgebra_predicate A P u).

  Definition Subalgebra : Algebra œÉ
    := BuildAlgebra carriers_subalgebra ops_subalgebra.
