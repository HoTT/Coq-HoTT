(** Recall that a "pointwise" functor is a functor [A·¥Æ ‚Üí C·¥∞] induced
    by functors [D ‚Üí B] and [A ‚Üí C].  Given two functors [D ‚Üí B] and a
    natural transformation between them, there is an induced natural
    transformation between the resulting functors between functor
    categories, and similarly for two functors [A ‚Üí C].  In this file,
    we construct these natural transformations. They will be used to
    construct the pointwise induced adjunction [FÀ£ ‚ä£ GÀ£] of an
    adjunction [F ‚ä£ G] for all categories [X]. *)

Local Open Scope morphism_scope.
Local Open Scope category_scope.
Local Open Scope functor_scope.
Local Open Scope natural_transformation_scope.

Section pointwise.
  Context `{Funext}.

  Variables C D C' D' : PreCategory.

  Local Ltac t :=
    path_natural_transformation; simpl;
    rewrite <- ?composition_of;
    try apply ap;
    first [ apply commutes
          | symmetry; apply commutes ].

  (** ** [TÀ£] for a natural transformation [T : F ‚Üí G] and a functor [x : C ‚Üí D] *)
  
H: Funext
C, D, C', D': PreCategory
F, G: Functor C D
T: NaturalTransformation F G
F': Functor C' D'
NaturalTransformation (pointwise F F')
  (pointwise G F')

  
H: Funext
C, D, C', D': PreCategory
F, G: Functor C D
T: NaturalTransformation F G
F': Functor C' D'
NaturalTransformation (pointwise F F')
  (pointwise G F')

    
H: Funext
C, D, C', D': PreCategory
F, G: Functor C D
T: NaturalTransformation F G
F': Functor C' D'
forall (s d : (D -> C')%category)
(m : morphism (D -> C') s d),
(F' o d oL T o (pointwise F F') _1 m)%morphism =
((pointwise G F') _1 m o (F' o s oL T))%morphism

    abstract t.
  Defined.

  (** ** [F·µÄ] for a natural transformation [T : F' ‚Üí G'] and a functor [F : C ‚Üí D] *)
  
H: Funext

F: Functor C D
F', G': Functor C' D'
T': NaturalTransformation F' G'
NaturalTransformation (pointwise F F')
  (pointwise F G')

  
H: Funext

NaturalTransformation (pointwise F F')

H: Funext
C, D, C', D': PreCategory
F: Functor C D
F', G': Functor C' D'
T': NaturalTransformation F' G'
forall (s d : (D -> C')%category)
(m : morphism (D -> C') s d),
(T' oR d oR F o (pointwise F F') _1 m)%morphism =
((pointwise F G') _1 m o (T' oR s oR F))%morphism

    abstract t.
  Defined.
End pointwise.
