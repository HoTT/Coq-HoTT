  (** A dependent diagram is said equifibered if all its fibers are equivalences. *)

  Class Equifibered {G : Graph} {D : Diagram G} (E : DDiagram D) := {
    isequifibered i j (g : G i j) (x : D i)
      :: IsEquiv (@arr _ E (i; x) (j; D _f g x) (g; idpath));
  }.
