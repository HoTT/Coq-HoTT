(** * Coproduct of natural transformations *)
Require Import Category.Sum Functor.Sum NaturalTransformation.Core.

Set Implicit Arguments.
Generalizable All Variables.

Section sum.

F, G: Core.Functor C D
F', G': Core.Functor C' D
T: NaturalTransformation F G
T': NaturalTransformation F' G'
NaturalTransformation (F + F') (G + G')

F, G: Core.Functor C D
F', G': Core.Functor C' D
T: NaturalTransformation F G
T': NaturalTransformation F' G'
forall (s d : (C + C')%category) (m : morphism (C + C') s d),
(match
   d as x
   return
     (morphism D (Core.object_of (F + F') x) (Core.object_of (G + G') x))
 with
 | Overture.inl c => T c
 | Overture.inr c' => T' c'
 end o Core.morphism_of (F + F') m)%morphism =
(Core.morphism_of (G + G') m
 o match
     s as x
     return
       (morphism D (Core.object_of (F + F') x) (Core.object_of (G + G') x))
   with
   | Overture.inl c => T c
   | Overture.inr c' => T' c'
   end)%morphism

    abstract (
        repeat (intros [] || intro); simpl;
        auto with natural_transformation
      ).
  Defined.
End sum.
