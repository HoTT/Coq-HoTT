(** This file implements algebra congruence relation. It serves as a
    universal algebra generalization of normal subgroup, ring ideal, etc.

    Congruence is used to construct quotients, in similarity with how
    normal subgroup and ring ideal are used to construct quotients. *)

Local Open Scope Algebra_scope.

Section congruence.
  Context {œÉ : Signature} (A : Algebra œÉ) (Œ¶ : forall s, Relation (A s)).

(** A finitary operation [f : A s1 * A s2 * ... * A sn -> A t]
    satisfies [OpCompatible f] iff

<<
      Œ¶ s1 x1 y1 * Œ¶ s2 x2 y2 * ... * Œ¶ sn xn yn
>>

    implies

<<
      Œ¶ t (f (x1, x2, ..., xn)) (f (y1, y2, ..., yn)).
>>

    The below definition generalizes this to infinitary operations.
*)

  Definition OpCompatible {w : SymbolType œÉ} (f : Operation A w) : Type
    := forall (a b : DomOperation A w),
       (forall i : Arity w, Œ¶ (sorts_dom w i) (a i) (b i)) ->
       Œ¶ (sort_cod w) (f a) (f b).

  Class OpsCompatible : Type
    := ops_compatible : forall (u : Symbol œÉ), OpCompatible u.#A.

    exact istrunc_forall.
  Defined.

  (** A family of relations [Œ¶] is a congruence iff it is a family of
      mere equivalence relations and [OpsCompatible A Œ¶] holds. *)

  Class IsCongruence : Type := Build_IsCongruence
   { is_mere_relation_cong :: forall (s : Sort œÉ), is_mere_relation (A s) (Œ¶ s)
   ; equiv_rel_cong :: forall (s : Sort œÉ), EquivRel (Œ¶ s)
   ; ops_compatible_cong :: OpsCompatible }.

  Global Arguments Build_IsCongruence {is_mere_relation_cong}
                                      {equiv_rel_cong}
                                      {ops_compatible_cong}.

{|
  is_mere_relation_cong := C1;

    by destruct (path_ishprop C1 D1),
                (path_ishprop C2 D2),

  Defined.

End congruence.

(** A homomorphism [f : forall s, A s -> B s] is compatible
    with a congruence [Œ¶] iff [Œ¶ s x y] implies [f s x = f s y]. *)

Definition HomCompatible {œÉ : Signature} {A B : Algebra œÉ}
  (Œ¶ : forall s, Relation (A s)) `{!IsCongruence A Œ¶}
  (f : forall s, A s -> B s) `{!IsHomomorphism f}
  : Type
  := forall s (x y : A s), Œ¶ s x y -> f s x = f s y.
