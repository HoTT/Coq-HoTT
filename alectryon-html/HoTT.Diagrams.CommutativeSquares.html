(** * Commutative squares *)

(** Commutative squares compose vertically.

      A --f--> B
      |    //  |
      h  comm  g
      |  //    |
      V //     V
      C --f'-> D
      |    //  |
      h' comm' g'
      |  //    |
      V //     V
      E --f''> F
*)

A, B, C, D, E, F: Type
f: A -> B
f': C -> D
h: A -> C
g: B -> D
comm: f' o h == g o f
f'': E -> F
h': C -> E
g': D -> F
comm': f'' o h' == g' o f'
f'' o (h' o h) == g' o g o f


A, B, C, D, E, F: Type
f: A -> B
f': C -> D
h: A -> C
g: B -> D
comm: f' o h == g o f
f'': E -> F
h': C -> E
g': D -> F
comm': f'' o h' == g' o f'
f'' o (h' o h) == g' o g o f

  
A, B, C, D, E, F: Type
f: A -> B
f': C -> D
h: A -> C
g: B -> D
comm: f' o h == g o f
f'': E -> F
h': C -> E
g': D -> F
comm': f'' o h' == g' o f'
x: A
f'' (h' (h x)) = g' (g (f x))

  apply ap, comm.
Defined.
