Definition pointwise_moveL_equiv {A A' B : Type} (R : Relation B)
  `{Reflexive _ R} `{Transitive _ R}
  (f : A <~> A') (P : A -> B) (Q : A' -> B)
  : relation_pointwise (fun _ => R) (P o f^-1) Q -> relation_pointwise (fun _ => R) P (Q o f)
  := pointwise_moveR_equiv (flip R) f Q P.
